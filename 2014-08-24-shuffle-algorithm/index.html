# !/usr/bin/env ruby
# coding: UTF-8

# ж–№жі•1
def shuffle01(arr)
    len = arr.length
    arr2 = []
    for i in 0..len-1
        index = Random.rand(0..arr.length-1)
        arr2 << arr[index]
        arr.delete_at(index)
    end
    return arr2
end

arr = ['a', 'b', 'c', 'd']
ret = shuffle01(arr)
p ret
