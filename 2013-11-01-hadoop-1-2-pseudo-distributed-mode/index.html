Hadoop1.2й…ҚзҪ®дјӘеҲҶеёғејҸ

еҸ‘иЎЁдәҺ 2013-11-01 | дә‘и®Ўз®— пјҢ hadoop

2013-11-01

дҪҝз”Ёзҡ„linux mint 15 64bitпјҢhadoopдҪҝз”Ёзҡ„жҳҜ1.2.1зүҲжң¬гҖӮ

1гҖҒи®ҫзҪ®ssh

е®үиЈ…sshзӣёе…іиҪҜд»¶еҢ…пјҡ 

sudo apt-get install openssh-client openssh-server

з„¶еҗҺдҪҝз”ЁдёӢйқўдёӨдёӘе‘Ҫд»Өд№ӢдёҖеҗҜеҠЁ/е…ій—ӯsshdпјҡ 

sudo /etc/init.d/ssh start|stop
sudo service ssh start|stop

иӢҘжҲҗеҠҹеҗҜеҠЁsshdпјҢжҲ‘д»¬иғҪзңӢеҲ°еҰӮдёӢзұ»дјјз»“жһңпјҡ 

$ ps -e | grep ssh
 2766 ?        00:00:00 ssh-agent
10558 ?        00:00:00 sshd

иҝҷж—¶еҖҷпјҢеҰӮжһңиҝҗиЎҢеҰӮдёӢsshзҷ»еҪ•жң¬жңәзҡ„е‘Ҫд»ӨпјҢдјҡжҸҗзӨәиҫ“е…ҘеҜҶз Ғпјҡ 

ssh localhost

зҺ°еңЁжҲ‘д»¬иҰҒеҒҡзҡ„е°ұжҳҜи®©е®ғдёҚйңҖиҰҒиҫ“е…ҘеҜҶз Ғпјҡ 

$ ssh-keygen -t rsa -P '' -f ~/.ssh/id_rsa  #дёҖдёӘз©әеҜҶз Ғзҡ„SSHеҜҶй’Ҙ
$ cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys

иҝҷж ·зҡ„иҜқе°ұиЎҢдәҶгҖӮеҰӮжһңд»ҚдёҚеҮ‘ж•ҲпјҢеҸҜиғҪжҳҜеҜҶй’Ҙж–Үд»¶зҡ„жқғйҷҗи®ҫзҪ®дёҚеҪ“гҖӮ

2гҖҒй…ҚзҪ®hadoop

е°Ҷhadoop-1.2.1и§ЈеҺӢеҲ°~/дёӢпјҢеҗҢж—¶еңЁ~/дёӢе»әз«Ӣзӣ®еҪ•hadoop-envпјҢз»§з»ӯеңЁhadoop-envдёӢе»әз«ӢеҰӮдёӢзҡ„зӣ®еҪ•з»“жһ„пјҡ

в”ңв”Җв”Җ dfs
в”Ӯ   в”ңв”Җв”Җ checkpoint1
в”Ӯ   в”ңв”Җв”Җ data1
в”Ӯ   в”ңв”Җв”Җ data2
в”Ӯ   в””в”Җв”Җ name1
в””в”Җв”Җ test
    в””в”Җв”Җ input

й…ҚзҪ®ж–Үд»¶hadoop-1.2.1/conf/core-site.xmlпјҡ 

<?xml version="1.0"?>
<?xml-stylesheet type="text/xsl" href="configuration.xsl"?>
<!-- Put site-specific property overrides in this file. -->
<configuration>
  <property>
    <name>fs.default.name</name>
    <value>hdfs://localhost:9000</value>
    <final>true</final>
  </property>
</configuration>

fs.default.nameжқҘжҢҮе®ҡHDFSзҡ„uriпјҢеҰӮжһңvalueдёӯжІЎжңүжҸҗдҫӣз«ҜеҸЈпјҢй»ҳи®Өдёә8020гҖӮ

й…ҚзҪ®ж–Үд»¶hadoop-1.2.1/conf/hdfs-site.xmlпјҡ 

<?xml version="1.0"?>
<?xml-stylesheet type="text/xsl" href="configuration.xsl"?>
<!-- Put site-specific property overrides in this file. -->
<configuration>
  <property>
    <name>dfs.name.dir</name>
    <value>~/hadoop-env/dfs/name1</value>
    <final>true</final>
  </property>
  <property>
    <name>dfs.data.dir</name>
    <value>~/hadoop-env/dfs/data1,~/hadoop-env/dfs/data2</value>
    <final>true</final>
  </property>
  <property>
    <name>fs.checkpoint.dir</name>
    <value>~/hadoop-env/dfs/checkpoint1</value>
    <final>true</final>
  </property>
</configuration>

dfs.name.dirжҢҮе®ҡnamenodeеӯҳеӮЁе…ғж•°жҚ®зҡ„зӣ®еҪ•пјҢеҸҜд»ҘжҢҮе®ҡеӨҡдёӘзӣ®еҪ•пјҢиҝҷдәӣзӣ®еҪ•иҰҒз”ЁйҖ—еҸ·еҲҶејҖпјӣdfs.data.dirжҢҮе®ҡdatanodeеӯҳж”ҫж•°жҚ®зҡ„зӣ®еҪ•пјҢеҸҜд»ҘжҢҮе®ҡеӨҡдёӘзӣ®еҪ•пјӣfs.checkpoint.dirжҢҮе®ҡиҫ…еҠ©namenodeеӯҳж”ҫжЈҖжҹҘзӮ№зҡ„зӣ®еҪ•гҖӮ

иҰҒжҳҜеӣҫдёӘж–№йқўзҡ„иҜқпјҢд№ҹеҸҜд»Ҙиҝҷж ·й…ҚзҪ®hadoop-1.2.1/conf/hdfs-site.xmlпјҡ

<configuration>
     <property>
         <name>dfs.replication</name>
         <value>1</value>
     </property>
</configuration>

й…ҚзҪ®ж–Үд»¶hadoop-1.2.1/conf/mapred-site.xmlпјҡ 

<?xml version="1.0"?>
<?xml-stylesheet type="text/xsl" href="configuration.xsl"?>
<!-- Put site-specific property overrides in this file. -->
<configuration>
  <property>
    <name>mapred.job.tracker</name>
    <value>localhost:9001</value>
    <final>true</final>
  </property>
</configuration>

еңЁ/etc/profileдёӯеҠ е…Ҙзӣёе…ізҺҜеўғеҸҳйҮҸпјҡ

# set hadoop 1.2.1
export HADOOP_HOME=/home/letian/hadoop-1.2.1  
export HADOOP_PREFIX=/home/letian/hadoop-1.2.1

з»Ҳз«Ҝдёӯжү§иЎҢдёӢйқўзҡ„е‘Ҫд»Өпјҡ

$ source /etc/profile
$ source ~/.bashrc

иӢҘйңҖиҰҒзҡ„иҜқпјҢжҠҠhadoop-1.2.1/binеҠ е…Ҙ$PATHгҖӮ

3гҖҒжөӢиҜ•

е…Ҳж јејҸеҢ–HDFSпјҡ 

./hadoop-1.2.1/bin/hadoop namenode -format

еҗҜеҠЁHDFSе’ҢMapReduceиҝӣзЁӢпјҡ 

$ ./hadoop-1.2.1/bin/start-dfs.sh
$ ./hadoop-1.2.1/bin/start-mapred.sh

еҗҜеҠЁж—¶еҖҷеҰӮжһңжҠҘй”ҷпјҢдҫӢеҰӮlocalhost: Error: JAVA_HOME is not set.пјҢйңҖиҰҒеңЁ./hadoop-1.2.1/conf/hadoop-env.shж–Үд»¶дёӯexport JAVA_HOMEпјҢдҫӢеҰӮпјҡ

export JAVA_HOME=~/jdk1.7.0_25

еҰӮдҪ•жЈҖжҹҘжҳҜеҗҰжҲҗеҠҹеҗҜеҠЁдәҶпјҡ
з¬¬дёҖз§Қж–№жі•жҳҜдҪҝз”Ёjpsе‘Ҫд»ӨпјҲJava Virtual Machine Process Status ToolпјүпјҢеә”иҜҘжңүеҰӮдёӢзұ»дјјиҫ“еҮәпјҡ 

$ jps
13592 DataNode
13728 SecondaryNameNode
13837 JobTracker
12864 NameNode
13955 TaskTracker
16069 Jps

з¬¬дәҢз§Қж–№жі•жҳҜжөҸи§ҲеҷЁзҷ»еҪ•http://localhost:50030жҹҘзңӢjobtrackerпјҢhttp://localhost:50070жҹҘзңӢnamenodeгҖӮеҰӮжһңеңЁдҪҝз”Ёзҝ»еўҷиҪҜд»¶пјҢйӮЈд№ҲеңЁи®ҝй—®ж—¶еҸҜиғҪдјҡжҠҘй”ҷпјҢжңҖз®ҖеҚ•зҡ„и§ЈеҶіж–№жі•жҳҜе…іжҺүзҝ»еўҷиҪҜд»¶гҖӮ
иҝҳжңүдёҖз§Қж–№жі•е°ұжҳҜжҹҘзңӢж—Ҙеҝ—ж–Үд»¶гҖӮ

зҺ°еңЁжҲ‘д»¬еңЁ~/hadoop-env/test/inputзӣ®еҪ•дёӢе»әз«ӢдёӨдёӘж–Үд»¶пјҡ 

$ echo "hello world" > test1.txt 
$ echo "hi,world" > test2.txt

жҠҠиҝҷдёӨдёӘж–Үд»¶еҜје…ҘHDFSпјҡ 

./hadoop-1.2.1/bin/hadoop dfs -put hadoop-env/test/input/  output/
