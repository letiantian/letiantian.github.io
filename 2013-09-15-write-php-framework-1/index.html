йҰ–е…Ҳеә”иҜҘиҜҙдёҖдёӢextractеҮҪж•°пјҢw3schoolиҝҷж ·иҜҙжҳҺиҜҘеҮҪж•°пјҡ

PHP extract() еҮҪж•°д»Һж•°з»„дёӯжҠҠеҸҳйҮҸеҜје…ҘеҲ°еҪ“еүҚзҡ„з¬ҰеҸ·иЎЁдёӯгҖӮ
еҜ№дәҺж•°з»„дёӯзҡ„жҜҸдёӘе…ғзҙ пјҢй”®еҗҚз”ЁдәҺеҸҳйҮҸеҗҚпјҢй”®еҖјз”ЁдәҺеҸҳйҮҸеҖјгҖӮ
з¬¬дәҢдёӘеҸӮж•° type з”ЁдәҺжҢҮе®ҡеҪ“жҹҗдёӘеҸҳйҮҸе·Із»ҸеӯҳеңЁпјҢиҖҢж•°з»„дёӯеҸҲжңүеҗҢеҗҚе…ғзҙ ж—¶пјҢextract() еҮҪж•°еҰӮдҪ•еҜ№еҫ…иҝҷж ·зҡ„еҶІзӘҒгҖӮ
жң¬еҮҪж•°иҝ”еӣһжҲҗеҠҹи®ҫзҪ®зҡ„еҸҳйҮҸж•°зӣ®гҖӮ

дҪңдёәеҸӮж•°зҡ„ж•°з»„дёӯзҡ„keyеә”иҜҘз¬ҰеҗҲеҸҳйҮҸе‘ҪеҗҚи§„иҢғпјҢдҫӢеҰӮextract([вҖҳthis->aaвҖҷ => вҖҳaaвҖҷ, вҖҳ1вҖҷ =?вҖҷbbвҖҷ])зҡ„иҝ”еӣһеҖје°ұдёә0гҖӮ

дёӢйқўжҳҜRenderзұ»дёӯrenderж–№жі•пјҡ

public function render($template_path, $data = null) 
{  
    $this->_template_path = Loader::get_instance()->root_dir() .'Template/' . $template_path .'.php';
    if(is_array($data)) {
        extract($data);
    }
    ob_start();
    echo eval('?>'.preg_replace("/;*\s*\?>/", "; ?>", 
        str_replace('<?=', '<?php echo ', file_get_contents($this->_template_path))));
    $this->_buffer = ob_get_contents();
    @ob_end_clean();
    $this->_content = $this->_content . $this->_buffer;
}

з”ұдәҺеңЁдёҡеҠЎйҖ»иҫ‘дёӯе·Із»Ҹе°ҶLoaderзұ»includeиҝӣжқҘпјҢжүҖд»ҘжҲ‘д»¬еҸҜд»ҘиҺ·еҫ—жЁЎжқҝж–Үд»¶зҡ„з»қеҜ№и·Ҝеҫ„$this->_template_pathпјӣд№ӢеҗҺextractе°Ҷ$dataеҲҶи§Јз”Ёд»Ҙеҗ‘и§Ҷеӣҫж–Үд»¶дј йҖ’ж•°жҚ®пјӣevalж–№жі•з”ЁжқҘдҪҝеҫ—жЁЎжқҝж–Үд»¶з”ҹж•ҲпјӣжЁЎжқҝж–Үд»¶з”ҹжҲҗзҡ„и§ҶеӣҫеҶ…е®№иҝҪеҠ еҲ°еҸҳйҮҸ$this->_bufferдёӯгҖӮеҫҲжҳҺжҳҫпјҢеңЁжҲ‘д»¬зҡ„жЁЎжқҝж–Үд»¶дёӯдёҚеә”иҜҘеҜ№$this->_bufferиҝӣиЎҢж“ҚдҪңгҖӮиҝҷйҮҢob_start()гҖҒ ob_get_contents()гҖҒob_end_clean()з”ЁдәҺжҺ§еҲ¶зј“еҶІеҢәпјҢе…·дҪ“иҜ·еҸӮиҖғPHPж–ҮжЎЈгҖӮ

renderж–№жі•д№ҹдёәжЁЎжқҝе®һзҺ°дәҶдёҖдёӘз®ҖеҚ•зҡ„зҹӯж Үи®°иҜӯжі•пјҲж— и®әPHPжң¬иә«жңүжІЎжңүејҖеҗҜзҹӯж Үи®°пјүпјҢдҫӢеҰӮ<?=$title;?>зӣёеҪ“дәҺ<? echo $title;?>гҖӮ
